1. Warmup-1 -- sleepIn

Solution: 

				
					function sleepIn(weekday, vacation){
						if(!weekday || vacation){
								return true;
  						}
  						return false;
					}

1.2 Warmup-1 -- sumDouble

Solution: 

				
					function sumDouble(a, b){
  						if(a == b){
    						return (a+b)*2;
  						}
  						return a+b;
					}

1.3 Warmup-1 -- diff21

Solution: 

				
					function diff21(n){
  						if(n > 21){
    						return (n-21)*2;
  						}
  						return 21-n;
					}

1.5 Warmup-1 -- makes10

Solution: 

				
					function makes10(a, b){
  						if(a == 10 || b == 10){
      						return true;
  						}
  						else if(a + b == 10){
    						return true;
  						}
  						return false;
					}

1.7 Warmup-1 -- posNeg

Solution: 

				
					function posNeg(a, b, negative){
  						if(!negative && a < 0 && b > 0){
    						return true;
  						}
  						else if(!negative && a > 0 && b < 0){
    						return true;
  						}
  						else if(negative && a < 0 && b < 0 ){
     						return true;
  						}
  						else{
    						return false;
  						}
					}

1.8 Warmup-1 -- notString

Solution: 

				
					function notString(str){
						if(str.substring(0,3) == "not"){
    						return str;
  						}
  						else{
   						return "not " + str;
						}
					}

1.9 Warmup-1 -- frontBack

Solution: 

				
					function frontBack(str){
  						if(str.length > 1){
    						return str.substring(str.length-1, str.length) + 
    						str.substring(1, str.length-1) + str.substring(0,1);
  						}
  						return str;
					}

					function front3(str){

    						return str+str+str;
  						}
  						return str.substring(0,3) + str.substring(0,3) + str.substring(0,3);
					}

1.11 Warmup-1 -- backAround

Solution: 

				
					function backAround(str){
  						if(str.length < 2){
    						return str + str + str;
  						}
 						return str.substring(str.length-1, str.length) + str + str.substring(str.length-1, str.length);
					}
