Warmup-1 -- sumDouble

Solution: 

				
	function sumDouble(a, b){
  		if(a == b){
    		return (a+b)*2
  		}
  		return a+b
	}

Warmup-1 -- diff21

Solution: 

				
	function diff21(n){
  		if(n > 21){
    		return (n-21)*2
  		}
  		return 21-n
	}
